100 /**
101 <b>Пример использования библиотеки cpp-mbee.</b>
102 \n
103 В данном примере микроконтроллер MSP430 выполняет роль пульта беспроводной системы управления освещением.
104 Для организации беспроводной сети применяются радиомодули MBee868-xx, работающие под управлением программного обеспечения SerialStar.
105 Все узлы сети должны иметь совпадающие сетевые настройки и эфирные параметры.
106 Радиомодуль, подключаемый к микроконтроллеру MSP430 пульта, должен работать в пакетном режиме (AP = 2). Битовая скорость на нем (устанавливается с помощью программы SysMCBootLoader)
107 должна совпадать со скоростью, определяемой параметром MBEE_UART_BITRATE.
108 Пульт осуществляет отправку AT-команд удаленному радиомодулю (target), адрес которого определяется константой TARGET_ADDRESS. Передача команд осуществляется кнопкой S1. Команды отправляются
109 циклически. Смысл каждой команды поясняется соответствующим комментарием.
110 Если проект компилируется с опцией ENABLE_CONSOLE, то второй модуль eUSCI, имеющийся в микроконтроллере MSP430 используется для выдачи диагностических сообщений, а также как
111 альтернативный управляющий интерфейс. Настройки параметров консоли осуществляются в файле control.h.
112 */
113 int main( void )
114 {
115  WDT_A_hold(WDT_A_BASE); //Отключаем встроенный watchdog-таймер во избежание его таймаута во время инициализации.
116  hardwareInit();
117  __enable_interrupt(); //Разрешаем прерывания, необходимые для работы модулей eUSCI в режиме UART.
118  serial.begin(MBEE_UART_BITRATE);
119  MBee.begin(serial);
120  #if defined(ENABLE_CONSOLE)
121  console.begin(CONSOLE_UART_BITRATE);
122  #endif
123 main_loop:
124  MBee.run();
125  #if defined(ENABLE_CONSOLE)
126  switch(console.run())
127  {
128  case '1':
129  targetSetup(); //Осуществляем начальную настройку неизменяющихся полей команды. Должна передана в начале работы перед!
130  break;
131  case '2':
132  targetDigitalOutSetHigh(); //Включаем дискретный цифровой выход на удаленном модуле.
133  break;
134  case '3':
135  targetDigitalOutSetLow(); //Выключаем дискретный цифровой выход на удаленном модуле.
136  break;
137  case '+':
138  brigthness += PWM_DUTY_CYCLE_MAX / BRIGTHNESS_NUMBER_OF_STEPS; //Увеличиваем яркость светодиода (рабочий цикл ШИМ) на одну ступень.
139  if(brigthness >= PWM_DUTY_CYCLE_MAX)
140  brigthness = PWM_DUTY_CYCLE_MAX;
141  targetPWMOutSetDutyCycle(brigthness);
142  break;
143  case '-':
144  brigthness -= PWM_DUTY_CYCLE_MAX / BRIGTHNESS_NUMBER_OF_STEPS; //Уменьшаем яркость светодиода (рабочий цикл ШИМ) на одну ступень.
145  if(brigthness <= 0)
146  brigthness = 0;
147  targetPWMOutSetDutyCycle(brigthness);
148  break;
149  default:
150  break;
151  }
152  #endif
153  if(TEST_EVENT(BUTTON_PRESSED_EVENT))
154  {
155  RESET_EVENT(BUTTON_PRESSED_EVENT);
156  switch(buttonMode)
157  {
158  case 0:
159  targetSetup(); //Осуществляем начальную на тройку неизменяющихся полей команды.
160  buttonMode = 1;
161  break;
162  case 1:
163  targetDigitalOutSetHigh(); //Включаем дискретный цифровой выход на удаленном модуле.
164  buttonMode = 2;
165  break;
166  case 2:
167  targetDigitalOutSetLow(); //Выключаем дискретный цифровой выход на удаленном модуле.
168  buttonMode = 3;
169  break;
170  case 3:
171  targetPWMOutSetDutyCycle(PWM_DUTY_CYCLE_MAX / 4); //Включаем ШИМ на 25%.
172  buttonMode = 4;
173  break;
174  case 4:
175  targetPWMOutSetDutyCycle(PWM_DUTY_CYCLE_MAX / 2); //Включаем ШИМ на 50%.
176  buttonMode = 5;
177  break;
178  case 5:
179  targetPWMOutSetDutyCycle(3 * PWM_DUTY_CYCLE_MAX / 4); //Включаем ШИМ на 75%.
180  buttonMode = 6;
181  break;
182  case 6:
183  targetPWMOutSetDutyCycle(PWM_DUTY_CYCLE_MAX); //Включаем ШИМ на 100%.
184  buttonMode = 7;
185  break;
186  case 7:
187  targetPWMOutSetDutyCycle(0); //Выключаем ШИМ.
188  buttonMode = 0;
189  break;
190  default:
191  break;
192  }
193  }
194  goto main_loop;
195 }

204 void targetSetup(void)
205 {
206  remoteAtRequest.setDefault(); //Приводим все поля команды для передачи к значениям "по умолчанию" для демонстрации наличия этой функции. Таким образом упрощается повоторное использование объекта для отправки последующих команд.
207  remoteAtRequest.setRemoteAddress(TARGET_ADDRESS); //Устанавливаем адрес удаленного модуля, которому будет передаваться команда.
208  remoteAtRequest.setCommand((uint8_t*)"R4"); //Устанавливаем код команды - вывод R4 (Сигнал DO2 на плате MB-Tag-1.1) переводим в режим первого ШИМ-выхода.
209  cmdParam[0] = IO_PWM1; //Устанавливаем значения параметра команды.
210  remoteAtRequest.setCommandValue(&cmdParam[0]); //Устанавливаем указатель на переменную с параметром команды.
211  remoteAtRequest.setCommandValueLength(1); //Длина поля параметра команды 1 байт.
212  sendCommand();
213 }

265 //*****************************************************************************
266 //
267 //! \brief Передача команды удаленному модулю.
268 //!
269 //! \return NONE
270 //
271 //*****************************************************************************
272 void sendCommand(void)
273 {
274  //Далее приводятся возможные опции отправки (см. описание опций в документации к библиотеке cpp-mbee.).
275  remoteAtRequest.setSaveChanges(false); //Изменения будут сразу применены без сохранения.
276  //remoteAtRequest.setCca(false); //Отключает режим CCA (Clear Channel Assessment) перед отправкой команды.
277  //remoteAtRequest.setAcknowledge(false); //Для повышения пропускной способности можно отключить подтверждение получения команды от удаленным модемом.
278  //remoteAtRequest.setApplyChanges(false); //Измененный параметр не будет применен и сохранен сразу после выполнения команды, а помещен в очередь.
279  //remoteAtRequest.setSleepingDevice(true); //Команда предназначается спящему модему.
280  MBee.send(remoteAtRequest); //Отправляем команду на удаленный модуль.
281  if(remoteAtRequest.getFrameId()) //Если frameID = 0, то локальный ответ не формируется и поэтому его не ждем.
282  {
283  if(MBee.readPacket(LOCAL_RESPONSE_TIMEOUT)) //Ждем локального ответа со статусом команды.
284  {
285  if(MBee.getResponse().getApiId() == TRANSMIT_STATUS_API_FRAME)//Проверяем, является ли принятый пакет локальным ответом на AT-команду удаленному модему.
286  {
287  MBee.getResponse().getTxStatusResponse(txStatus); //Получаем объект с локальным статусом отправки.
288  #if defined(ENABLE_CONSOLE)
289  if(txStatus.isSuccess())
290  {
291  if(remoteAtRequest.getSleepingDevice())
292  {
293  console.print("Command is successful buffered for sleeping target with address: ");
294  console.println(TARGET_ADDRESS, 16);
295  }
296  else
297  {
298  console.print("Command is successful transmitted to the target with address: ");
299  console.println(TARGET_ADDRESS, 16);
300  }
301  }
302  else if(txStatus.getStatus() == ERROR_COMMAND_STATUS)
303  {
304  console.println("Error! Unsufficient memory for packet.");
305  }
306  else if(txStatus.getStatus() == INVALID_CODE_COMMAND_STATUS)
307  {
308  console.println("Error! Invalid command code.");
309  }
310  else if(txStatus.getStatus() == INVALID_PARAMETER_COMMAND_STATUS)
311  {
312  console.println("Error! Invalid parameter value.");
313  }
314  else if(txStatus.getStatus() == TX_FAILURE_COMMAND_STATUS)
315  {
316  console.println("Warning! Command is not transmitted due to CCA failure.");
317  }
318  #endif
319  }
320  }
321  }
322  if((remoteAtRequest.getFrameId() == 0) || (txStatus.isSuccess() && remoteAtRequest.getSleepingDevice() == false)) //Ждем ответного пакета от удаленного модема только если локальный ответ выключен с помощью frameId = 0, команда отправлена(буферизирована) и не предназначается спящему модему.
323  {
324  if(MBee.readPacket(ACKNOWLEDGE_TIMEOUT)) //Ждем ответа от удаленного модуля со статусом команды.
325  {
326  if(MBee.getResponse().getApiId() == REMOTE_AT_COMMAND_RESPONSE_API_FRAME) //Проверяем, не является ли полученный фрейм ответом на команду удаленным модемом.
327  {
328  MBee.getResponse().getRemoteAtCommandResponse(remoteAtResponse);
329  if(remoteAtResponse.isOk())
330  {
331  ledBlinking(GREEN, 2); //Команда успешно доставлена на удаленый модем.
332  #if defined(ENABLE_CONSOLE)
333  console.print("Command is successful delivered to the target with address: "); //Получение команды подтверждено удаленным модулем.
334  console.println(TARGET_ADDRESS, 16);
335  #endif
336  }
337  }
338  }
339  else
340  {
341  ledBlinking(RED, 3); //Нет ответа на команду от удаленного модема.
342  #if defined(ENABLE_CONSOLE)
343  console.println("No response from the target with address: "); //Нет ответа от удаленного модуля.
344  #endif
345  }
346  }
347 }

351 //! \brief Инициализация всех линий ввода/вывода и периферийных модулей.

356 void hardwareInit(void)
357 {
358  GPIO_setAsPeripheralModuleFunctionInputPin(GPIO_PORT_XIN, GPIO_PIN_XIN + GPIO_PIN_XOUT, GPIO_FUNCTION_XIN); //Настраиваем выводы, к которым подключен кварцевый резонатор 32768 Гц.
359  GPIO_setAsPeripheralModuleFunctionInputPin(MBEE_UART_PORT_NUM, MBEE_UART_RX_BIT_NUM, GPIO_FUNCTION_UCA0RXD); //Настройка выводов, используемых модулем UART для соединения с радиомодулем MBee.
360  GPIO_setAsPeripheralModuleFunctionOutputPin(MBEE_UART_PORT_NUM, MBEE_UART_TX_BIT_NUM, GPIO_FUNCTION_UCA0TXD);
361  #if defined(ENABLE_CONSOLE)
362  GPIO_setAsPeripheralModuleFunctionInputPin(CONSOLE_UART_PORT_NUM, CONSOLE_UART_RX_BIT_NUM, GPIO_FUNCTION_CONSOLE_UCAxRXD); //Настройка выводов, используемых модулем UART для организации консоли.
363  GPIO_setAsPeripheralModuleFunctionOutputPin(CONSOLE_UART_PORT_NUM, CONSOLE_UART_TX_BIT_NUM, GPIO_FUNCTION_CONSOLE_UCAxTXD);
364  #endif
365  GPIO_setOutputLowOnPin(GPIO_PORT_LED1, GPIO_PIN_LED1); //Настройка вывода, управляющего первым светодиодом.
366  GPIO_setAsOutputPin(GPIO_PORT_LED1, GPIO_PIN_LED1);
367  GPIO_setOutputLowOnPin(GPIO_PORT_LED2, GPIO_PIN_LED2); //Настройка вывода, управляющего первым светодиодом.
368  GPIO_setAsOutputPin(GPIO_PORT_LED2, GPIO_PIN_LED2);
369  GPIO_setAsInputPinWithPullUpResistor(GPIO_PORT_S1, GPIO_PIN_S1); //Настраиваем, вывод, к которому подключена первая кнопка.
370  GPIO_enableInterrupt(GPIO_PORT_S1, GPIO_PIN_S1);
371  GPIO_selectInterruptEdge(GPIO_PORT_S1, GPIO_PIN_S1, GPIO_HIGH_TO_LOW_TRANSITION);
372  GPIO_clearInterrupt(GPIO_PORT_S1, GPIO_PIN_S1);
373 
374  PMM_unlockLPM5(); //Активируем настройки портов.
375  CS_turnOnXT1LFWithTimeout(CS_XT1_DRIVE_3 , CS_XT1_TIMEOUT); //Включение кварцевого генератора с самым высоким усилением и, соответственно, с самой высокой помехоустойчивостью и более высоким потреблением.
376  CS_initClockSignal(CS_FLLREF, CS_XT1CLK_SELECT, CS_CLOCK_DIVIDER_1 ); //В качестве источника тактового сигнала DCO выбираем кварцевый генератор 32768 Гц без входного деления частоты.
377  CS_initFLL(CS_MCLK_DESIRED_FREQUENCY_IN_KHZ, CS_MCLK_FLLREF_RATIO); //Устанавливаем требуемую частоту сигнала DCO. Эта же функция выбирает DCO как источник и MCLK и SMCLK. Для работы библиотеки cpp-mbee требуется,чтобы тактовая частота SMCLK была равной 8 МГц.
378  CS_initClockSignal(CS_ACLK, CS_XT1CLK_SELECT, CS_CLOCK_DIVIDER_1); //Для примера для ACLK выбираем кварцевый генератор 32768 Гц.
379 
380  //Настройка таймера Timer1_B3, используемого для формирования задержек. Настройка осуществляется только частично так как старт таймера и часть настроек будет выполняться непосредственно в функциях задержки.
381  Timer_B_initUpModeParam timerbConfig = {0};
382  timerbConfig.clockSource = TIMER_B_CLOCKSOURCE_SMCLK;
383  timerbConfig.clockSourceDivider = TIMER_B_CLOCKSOURCE_DIVIDER_8; //Тактовую частоту Timer1_B3 устанавливаем равной 1 МГц.
384  Timer_B_initUpMode(TIMER_B1_BASE, &timerbConfig);
385 }

387 //*****************************************************************************
388 //
389 //! \brief Включает требуемый светодиод заданное число раз.
390 //!
391 //! \param led Код цвета светодиода.
392 //! \param count Число включений светодиода.
393 //!
394 //! \return NONE
395 //
396 //*****************************************************************************
397 void ledBlinking(uint8_t led, uint8_t count)
398 {
399  while(count--)

425 //!
426 //! \param us Микросекунды.
427 //!
428 //! \return NONE

430 //*****************************************************************************
431 void delayUs(uint16_t us)
432 {
433  uint16_t intState;
434  uint16_t count = us/MAX_DELAY_US;
435  if(us)
436  {
437  TB1CTL &= ~(MC_1 | MC_2); //Останавливаем таймер на случай, если он был включен.
438  TB1CTL |= TBCLR; //Сбрасываем TB0 и делители.
439  TB1CCR0 = (MAX_DELAY_US - 1);
440  TB1CTL |= MC__UP; //Запускаем таймер в UP mode.
441  TB1CCTL0 &= ~CCIFG;
442  do
443  {
444  ENTER_CRITICAL_SECTION(intState);
445  while(!(TB1CCTL0 & CCIFG));
446  TB1CCTL0 &= ~CCIFG;
447  EXIT_CRITICAL_SECTION(intState);
448  }while(count--);
449  TB1CTL &= ~(MC_1 | MC_2);
450  }
451 }
