128 void Serial::write(uint8_t byte)
129 {
130  interruptState_t intState;
131  ENTER_CRITICAL_SECTION(intState, USART1_IRQn);
132  uint32_t temp = txBufferPointer;
133  uint32_t offset = temp + txBufferCounter++;
134  if(offset >= UART_TX_BUFFER_SIZE)
135  {
136  offset -= UART_TX_BUFFER_SIZE;
137  }
138  txBuffer[offset++] = byte;
139  EXIT_CRITICAL_SECTION(intState, USART1_IRQn);
140  if(offset == UART_TX_BUFFER_SIZE)
141  {
142  offset = 0u;
143  }
144  HAL_UART_Transmit_IT(&huart1, &byte, 1);
145 }

165 void MBEE_HAL_UART_TxCpltCallback(UART_HandleTypeDef *UartHandle)
166 {
167  txBufferCounter--;
168  if(txBufferCounter)
169  {
170  if(HAL_UART_Transmit_IT(UartHandle, &txBuffer[++txBufferPointer], 1) != HAL_OK)
171  {
172  Error_Handler();
173  }
174  if(txBufferPointer == UART_TX_BUFFER_SIZE)
175  {
176  txBufferPointer = 0u;
177  }
178  }
179 }

181 void MBEE_HAL_UART_RxCpltCallback(UART_HandleTypeDef *UartHandle)
182 {
183  uint32_t temp;
184  uint32_t offset;
185  temp = rxBufferPointer;
186  offset = temp + rxBufferCounter;
187  if(offset >= UART_RX_BUFFER_SIZE)
188  {
189  offset -= UART_RX_BUFFER_SIZE;
190  }
191  rxBuffer[offset] = RxByte;
192  if(rxBufferCounter == UART_RX_BUFFER_SIZE)
193  {
194  rxBufferPointer++;
195  if(rxBufferPointer == UART_RX_BUFFER_SIZE)
196  {
197  rxBufferPointer = 0u;
198  }
199  }
200  else
201  {
202  rxBufferCounter++;
203  }
204  HAL_UART_Receive_IT(UartHandle, &RxByte, 1);
205 }
