87 /**
88 <b>Пример использования библиотеки cpp-mbee.</b>
89 \n
90 В данном примере микроконтроллер MSP430 работает в качестве автономного сигнализатора беспроводной сети тревожных датчиков.
91 В качестве беспроводных датчиков могут быть использованы датчики протечки, пожарно-охраннные извещатели, датчики газа, разбития стекла и тому подобные устройства
92 с цифровым выходом.
93 Для организации беспроводной сети применяются радиомодули MBee868-xx, работающие под управлением программного обеспечения SerialStar.
94 Все узлы сети должны иметь совпадающие сетевые настройки и эфирные параметры.
95 Модули, подключаемые к датчикам, должны иметь собственные адреса (параметр MY), лежащие в диапазоне SENSOR_ADDRESSES_START..SENSOR_ADDRESSES_END. Обычно такие датчики имеют автономное питание и поэтому должны
96 большую часть времени находится в спящем режиме. Отправка ими пакета, содержащего данные о состоянии линий ввода/вывода может осуществляться по срабатыванию датчика (параметр SM, должен быть установлен, равным 1),
97 по внутреннему таймеру(SM = 4) или в комбинированном режиме (SM = 5). Описание режимов сна находится в техническом описании программного обеспечения SerialStar.
98 Радиомодуль, подключаемый к микроконтроллеру MSP430 сигнализатора, должен работать в пакетном режиме (AP = 2). Битовая скорость на нем (устанавливается с помощью программы SysMCBootLoader)
99 должна совпадать со скоростью, определяемой параметром MBEE_UART_BITRATE.
100 Сигнализатор осуществляет прием пакетов (пакет 0x83) с данными о состоянии линий ввода/вывода от датчиков и фильтрует их на принадлежность заданному диапазону.
101 Если в пакете содержится информация о состоянии вывода радиомодя, определяемого константой ALARM_INPUT и если данный вывод используется как цифровой вход, то Сигнализатор
102 управляет своим выводом, задаваемым параметрами GPIO_PORT_LED1 и GPIO_PIN_LED1 (файл board.h) в зависимости от состояния входа ALARM_INPUT.
103 Тревожный сигнал на выходе сигнализатора может быть временно (до прихода очередного пакета) выключен кнопкой S1, настройки которой также определяются в файле board.h.
104 Если проект компилируется с опцией ENABLE_CONSOLE, то второй модуль eUSCI, имеющийся в микроконтроллере MSP430 используется для выдачи диагностических сообщений, а также как
105 альтернативный управляющий интерфейс. Отправив на UART консоли символ '1' можно временно выключить выходной сигнал также, как это происходит при нажатии кнопки S1.
106 Настройки параметров консоли осуществляются в файле alarm.h.
107 */
108 int main( void )
109 {
110  WDT_A_hold(WDT_A_BASE); //Отключаем встроенный watchdog-таймер во избежание его таймаута во время инициализации.
111  hardwareInit();
112  __enable_interrupt(); //Разрешаем прерывания, необходимые для работы модулей UART.
113  serial.begin(MBEE_UART_BITRATE);
114  MBee.begin(serial);
115  #if defined(ENABLE_CONSOLE)
116  console.begin(CONSOLE_UART_BITRATE);
117  #endif
118  MBee.onRxIoSampleResponse(ioSamplePacketReceived); //Регистрация callback-функции для приема пакетов 0x83 с состоянием линий ввода-вывода.
119 main_loop:
120  MBee.run();
121  #if defined(ENABLE_CONSOLE)
122  switch(console.run())
123  {
124  case '1':
125  switchOffAlarm();
126  break;
127  default:
128  break;
129  }
130  #endif
131  if(TEST_EVENT(BUTTON_PRESSED_EVENT))
132  {
133  RESET_EVENT(BUTTON_PRESSED_EVENT);
134  switchOffAlarm();
135  }
136  if(TEST_EVENT(IO_SAMPLE_PACKET_RECEIVED_EVENT))
137  {
138  RESET_EVENT(IO_SAMPLE_PACKET_RECEIVED_EVENT);
139  parseIoSamplePacket();
140  }
141  goto main_loop;
142 }

164 //*****************************************************************************
165 //
166 //! \brief Разбор принятого пакета 0x83.
167 //!
168 //! \return NONE
169 //
170 //*****************************************************************************
171 void parseIoSamplePacket(void)
172 {
173  uint8_t sampleNumber;
174  //Прием каждого пакета индицируем светодиодом.
175  ledBlinking(GREEN, 1);
176  if((packet.getRemoteAddress() >= SENSOR_ADDRESSES_START) && (packet.getRemoteAddress() <= SENSOR_ADDRESSES_END)) //Фильтруем входящий пакет по диапазону адресов.
177  {
178  if(packet.getSampleSize()) //Содержится ли в принятом пакете информация о состоянии линий ввода/вывода удаленного модема?
179  {
180  sampleNumber = packet.getSampleNumber(ALARM_INPUT);
181  if(sampleNumber) //Есть ли в принятом пакете информация о состоянии вывода радиомодуля, к которому подключен цифровой датчик и какой номер выборки он имеет?
182  {
183  if((packet.getMode(sampleNumber) & 0x7F) == IO_DIGITAL_INPUT) //Настроен ли данный вывод как цифровой вход?
184  {
185  if(packet.getDigital(sampleNumber) == LOW)
186  {
187  GPIO_setOutputHighOnPin(GPIO_PORT_LED1, GPIO_PIN_LED1); //Вход на удаленном модуле замкнут. Включаем цифровой выход "Тревога".
188  }
189  else
190  {
191  GPIO_setOutputLowOnPin(GPIO_PORT_LED1, GPIO_PIN_LED1); //Вход на удаленном модуле разомкнут. Выключаем цифровой выход "Тревога".
192  }
193  }
194  }
195  }
196  }
197  #if defined(ENABLE_CONSOLE)
198  console.printIoSamplePacket(packet);
199  #endif
200 }

229  CS_initClockSignal(CS_FLLREF, CS_XT1CLK_SELECT, CS_CLOCK_DIVIDER_1 ); //В качестве источника тактового сигнала DCO выбираем кварцевый генератор 32768 Гц без входного деления частоты.
230  CS_initFLL(CS_MCLK_DESIRED_FREQUENCY_IN_KHZ, CS_MCLK_FLLREF_RATIO); //Устанавливаем требуемую частоту сигнала DCO. Эта же функция выбирает DCO как источник и MCLK и SMCLK. Для работы библиотеки cpp-mbee требуется,чтобы тактовая частота SMCLK была равной 8 МГц.
231  CS_initClockSignal(CS_ACLK, CS_XT1CLK_SELECT, CS_CLOCK_DIVIDER_1); //Для примера для ACLK выбираем кварцевый генератор 32768 Гц.
232 
233  //Настройка таймера Timer1_B3, используемого для формирования задержек. Настройка осуществляется только частично так как старт таймера и часть настроек будет выполняться непосредственно в функциях задержки.
234  Timer_B_initUpModeParam timerbConfig = {0};
235  timerbConfig.clockSource = TIMER_B_CLOCKSOURCE_SMCLK;
236  timerbConfig.clockSourceDivider = TIMER_B_CLOCKSOURCE_DIVIDER_8; //Тактовую частоту Timer1_B3 устанавливаем равной 1 МГц.
237  Timer_B_initUpMode(TIMER_B1_BASE, &timerbConfig);
238 }

272 //*****************************************************************************
273 //
274 //! \brief Задержка микроcекундного диапазона.
275 //!
276 //! Обеспечивает thread-безопасную задержку миллисекундного диапазона. Использует
277 //! таймер Timer1_B3.
278 //!
279 //! \param us Микросекунды.
280 //!
281 //! \return NONE
282 //
283 //*****************************************************************************
284 void delayUs(uint16_t us)
285 {
286  uint16_t intState;
287  uint16_t count = us/MAX_DELAY_US;
288  if(us)
289  {
290  TB1CTL &= ~(MC_1 | MC_2); //Останавливаем таймер на случай, если он был включен.
291  TB1CTL |= TBCLR; //Сбрасываем TB0 и делители.
292  TB1CCR0 = (MAX_DELAY_US - 1);
293  TB1CTL |= MC__UP; //Запускаем таймер в UP mode.
294  TB1CCTL0 &= ~CCIFG;
295  do
296  {
297  ENTER_CRITICAL_SECTION(intState);
298  while(!(TB1CCTL0 & CCIFG));
299  TB1CCTL0 &= ~CCIFG;
300  EXIT_CRITICAL_SECTION(intState);
301  }while(count--);
302  TB1CTL &= ~(MC_1 | MC_2);
303  }
304 }
