163 void Console::print(uint32_t parameter, uint8_t radix)
164 {
165  uint8_t buffer[10 + 1]; //Для передачи 4 байт в 10-тичном виде требуется 10 символов + 1 терминирующий символ 0x00.
166  switch(radix)
167  {
168  case 10:
169  ultoa(parameter, buffer, radix);
170  break;
171  case 16:
172  *buffer = '0';
173  *(buffer + 1) = 'x';
174  ultoa(parameter, buffer + 2, radix);
175  break;
176  }
177  uartTxWait(buffer, strlen((char*)buffer));
178 }

201 uint8_t* Console::ultoa(uint32_t num, uint8_t *buf, uint8_t radix)
202 {
203  uint8_t temp[33];
204  int8_t index = 0;
205  uint8_t digit;
206  uint8_t stringIndex = 0;
207  do
208  {
209  digit = num % radix;
210  if(digit < 10)
211  temp[index++] = digit + '0';
212  else
213  temp[index++] = digit - 10 + 'A';
214  num = (num) / radix;
215  }while(num > 0);
216  index--;
217  while (index >=0)
218  buf[stringIndex++] = temp[index--];
219  buf[stringIndex] = 0;
220  return((uint8_t*)buf);
221 }
