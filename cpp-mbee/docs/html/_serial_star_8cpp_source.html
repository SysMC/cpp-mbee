480 /*********************************************************************
481 * Private методы класса RxIoSampleResponse.
482 *********************************************************************/
483 void RxIoSampleResponse::getNextSample(uint8_t &alias)
484 {
485  switch(getData()[alias] & 0x7F) //Сбрасываем старший бит, в котором передается информация о текущем состоянии цифровой линии.
486  {
487  case IO_ADC:
488  alias += 4;
489  break;
490  case IO_DIGITAL_INPUT:
491  case IO_DIGITAL_OUTPUT_LO:
492  case IO_DIGITAL_OUTPUT_HI:
493  case IO_WAKEUP_INPUT_FALLING_EDGE:
494  case IO_WAKEUP_INPUT_RISING_EDGE:
495  alias += 2;
496  break;
497  case IO_COUNTER_INPUT1:
498  case IO_COUNTER_INPUT2:
499  alias += 6;
500  break;
501  }
502 }

504 uint8_t RxIoSampleResponse::getSample(uint8_t number)
505 {
506  uint8_t count = 1;
507  uint8_t i = 1;
508  uint8_t & iRef = i;
509  while(i < getDataLength())
510  {
511  if(count == number)
512  break;
513  getNextSample(iRef);
514  count++;
515  }
516  return i;
517 }

795 uint8_t RemoteAtCommandRequest::getFrameData(uint8_t pos)
796 {
797  if(pos == 0)
798  {
799  return (_remoteAddress >> 8) & 0xff;
800  }
801  else if(pos == 1)
802  {
803  return _remoteAddress & 0xff;
804  }
805  else if(pos == 2)
806  {
807  return ((_acknowledge ? 0: BV(ACKNOWLEDGE_DISABLE_COMMAND_OPTIONS_BIT)) +\
808  (_applyChanges ? BV(APPLY_CHANGES_COMMAND_OPTION_BIT): 0) +\

1001 /*********************************************************************
1002 * Переопределенные Public методы класса SerialStar, наследуемые из MBee868
1003 *********************************************************************/
1004 void SerialStar::readPacket()
1005 {
1006  if(_response.isAvailable() || _response.isError())
1007  resetResponse(); //Сброс предыдущих сообщений.
1008  while(available())
1009  {
1010  b = read();
1011  if(_pos > 0 && b == START_BYTE && ATAP == 2)
1012  {
1013  //Обнаружен стартовый байт до окончания обработки предыдущего пакета. Сбрасываем предыдущее сообщение и начинаем разбор заново.
1014  _response.setErrorCode(UNEXPECTED_START_BYTE);
1015  return;
1016  }
1017  if(_pos > 0 && b == ESCAPE)
1018  {
1019  if(available())
1020  {
1021  b = read();
1022  b = 0x20 ^ b; //Извлекаем символ из escape-последовательности.
1023  }
1024  else
1025  {
1026  _escape = true; //Ставим флаг того, что следующий байт будет являться вторым символом из escape-последовательности.
1027  continue;
1028  }
1029  }
1030  if(_escape == true)
1031  {
1032  b = 0x20 ^ b;
1033  _escape = false; //Предыдущий байт был escape-символом. Извлекаем символ из escape-последовательности.
1034  }
1035  if(_pos >= API_ID_INDEX) //Контрольную сумму считаем начиная с байта типа API-фрейма.
1036  _checksumTotal+= b;
1037 
1038  //Конечный автомат парсера.
1039  switch(_pos)
1040  {
1041  case 0:
1042  if(b == START_BYTE)
1043  _pos++;
1044  break;
1045  case 1:
1046  _response.setMsbLength(b);
1047  _pos++;
1048  break;
1049  case 2:
1050  _response.setLsbLength(b);
1051  _pos++;
1052  break;
1053  case 3:
1054  _response.setApiId(b);
1055  _pos++;
1056  break;
1057  default:
1058  //Далее разбираются байты, следующие за 4-ым.
1059  if(_pos > MAX_FRAME_DATA_SIZE)
1060  {
1061  _response.setErrorCode(PACKET_EXCEEDS_BYTE_ARRAY_LENGTH); //Длина пакета превышает максимально допустимую.
1062  return;
1063  }
1064  //Проверяем, не достигли ли еще конца поля данных.
1065  if(_pos == (_response.getPacketLength() + 3))
1066  {
1067  if((_checksumTotal & 0xff) == 0xff)
1068  {
1069  _response.setChecksum(b);
1070  _response.setAvailable(true);
1071  _response.setErrorCode(NO_ERROR_IN_FRAME);
1072  }
1073  else
1074  _response.setErrorCode(CHECKSUM_FAILURE); //Ошибка контрольной суммы.
1075  _response.setFrameLength(_pos - 4); //Длина поля данных не включает также поле Frame Type.
1076  _pos = 0;
1077  return;
1078  }
1079  else
1080  {
1081  _response.getFrameData()[_pos - 4] = b; //Добавляем считанный символ в поле данных.
1082  _pos++;
1083  }
1084  }
1085  }
1086 }
