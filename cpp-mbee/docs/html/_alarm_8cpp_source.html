171 void parseIoSamplePacket(void)
172 {
173  uint8_t sampleNumber;
174  //Прием каждого пакета индицируем светодиодом.
175  ledBlinking(GREEN, 1);
176  if((packet.getRemoteAddress() >= SENSOR_ADDRESSES_START) && (packet.getRemoteAddress() <= SENSOR_ADDRESSES_END)) //Фильтруем входящий пакет по диапазону адресов.
177  {
178  if(packet.getSampleSize()) //Содержится ли в принятом пакете информация о состоянии линий ввода/вывода удаленного модема?
179  {
180  sampleNumber = packet.getSampleNumber(ALARM_INPUT);
181  if(sampleNumber) //Есть ли в принятом пакете информация о состоянии вывода радиомодуля, к которому подключен цифровой датчик и какой номер выборки он имеет?
182  {
183  if((packet.getMode(sampleNumber) & 0x7F) == IO_DIGITAL_INPUT) //Настроен ли данный вывод как цифровой вход?
184  {
185  if(packet.getDigital(sampleNumber) == LOW)
186  {
187  GPIO_setOutputHighOnPin(GPIO_PORT_LED1, GPIO_PIN_LED1); //Вход на удаленном модуле замкнут. Включаем цифровой выход "Тревога".
188  }
189  else
190  {
191  GPIO_setOutputLowOnPin(GPIO_PORT_LED1, GPIO_PIN_LED1); //Вход на удаленном модуле разомкнут. Выключаем цифровой выход "Тревога".
192  }
193  }
194  }
195  }
196  }
197  #if defined(ENABLE_CONSOLE)
198  console.printIoSamplePacket(packet);
199  #endif
200 }

209 void hardwareInit(void)
210 {
211  GPIO_setAsPeripheralModuleFunctionInputPin(GPIO_PORT_XIN, GPIO_PIN_XIN + GPIO_PIN_XOUT, GPIO_FUNCTION_XIN); //Настраиваем выводы, к которым подключен кварцевый резонатор 32768 Гц.
212  GPIO_setAsPeripheralModuleFunctionInputPin(MBEE_UART_PORT_NUM, MBEE_UART_RX_BIT_NUM, GPIO_FUNCTION_UCA0RXD); //Настройка выводов, используемых модулем UART для соединения с радиомодулем MBee.
213  GPIO_setAsPeripheralModuleFunctionOutputPin(MBEE_UART_PORT_NUM, MBEE_UART_TX_BIT_NUM, GPIO_FUNCTION_UCA0TXD);
214  #if defined(ENABLE_CONSOLE)
215  GPIO_setAsPeripheralModuleFunctionInputPin(CONSOLE_UART_PORT_NUM, CONSOLE_UART_RX_BIT_NUM, GPIO_FUNCTION_CONSOLE_UCAxRXD); //Настройка выводов, используемых модулем UART для организации консоли.
216  GPIO_setAsPeripheralModuleFunctionOutputPin(CONSOLE_UART_PORT_NUM, CONSOLE_UART_TX_BIT_NUM, GPIO_FUNCTION_CONSOLE_UCAxTXD);
217  #endif
218  GPIO_setOutputLowOnPin(GPIO_PORT_LED1, GPIO_PIN_LED1); //Настройка вывода, управляющего первым светодиодом.
219  GPIO_setAsOutputPin(GPIO_PORT_LED1, GPIO_PIN_LED1);
220  GPIO_setOutputLowOnPin(GPIO_PORT_LED2, GPIO_PIN_LED2); //Настройка вывода, управляющего первым светодиодом.
221  GPIO_setAsOutputPin(GPIO_PORT_LED2, GPIO_PIN_LED2);
222  GPIO_setAsInputPinWithPullUpResistor(GPIO_PORT_S1, GPIO_PIN_S1); //Настраиваем, вывод, к которому подключена первая кнопка.
223  GPIO_enableInterrupt(GPIO_PORT_S1, GPIO_PIN_S1);
224  GPIO_selectInterruptEdge(GPIO_PORT_S1, GPIO_PIN_S1, GPIO_HIGH_TO_LOW_TRANSITION);
225  GPIO_clearInterrupt(GPIO_PORT_S1, GPIO_PIN_S1);
226 
227  PMM_unlockLPM5(); //Активируем настройки портов.
228  CS_turnOnXT1LFWithTimeout(CS_XT1_DRIVE_3 , CS_XT1_TIMEOUT); //Включение кварцевого генератора с самым высоким усилением и, соответственно, с самой высокой помехоустойчивостью и более высоким потреблением.
229  CS_initClockSignal(CS_FLLREF, CS_XT1CLK_SELECT, CS_CLOCK_DIVIDER_1 ); //В качестве источника тактового сигнала DCO выбираем кварцевый генератор 32768 Гц без входного деления частоты.
230  CS_initFLL(CS_MCLK_DESIRED_FREQUENCY_IN_KHZ, CS_MCLK_FLLREF_RATIO); //Устанавливаем требуемую частоту сигнала DCO. Эта же функция выбирает DCO как источник и MCLK и SMCLK. Для работы библиотеки cpp-mbee требуется,чтобы тактовая частота SMCLK была равной 8 МГц.
231  CS_initClockSignal(CS_ACLK, CS_XT1CLK_SELECT, CS_CLOCK_DIVIDER_1); //Для примера для ACLK выбираем кварцевый генератор 32768 Гц.
232 
233  //Настройка таймера Timer1_B3, используемого для формирования задержек. Настройка осуществляется только частично так как старт таймера и часть настроек будет выполняться непосредственно в функциях задержки.
234  Timer_B_initUpModeParam timerbConfig = {0};
235  timerbConfig.clockSource = TIMER_B_CLOCKSOURCE_SMCLK;
236  timerbConfig.clockSourceDivider = TIMER_B_CLOCKSOURCE_DIVIDER_8; //Тактовую частоту Timer1_B3 устанавливаем равной 1 МГц.
237  Timer_B_initUpMode(TIMER_B1_BASE, &timerbConfig);
238 }

250 void ledBlinking(uint8_t led, uint8_t count)
251 {
252  while(count--)
253  {
254  switch(led)
255  {
256  case RED:
257  GPIO_setOutputHighOnPin(GPIO_PORT_LED1, GPIO_PIN_LED1);
258  delayMs(LED_ON_TIME);
259  GPIO_setOutputLowOnPin(GPIO_PORT_LED1, GPIO_PIN_LED1);
260  delayMs(LED_ON_TIME);
261  break;
262  case GREEN:
263  GPIO_setOutputHighOnPin(GPIO_PORT_LED2, GPIO_PIN_LED2);
264  delayMs(LED_ON_TIME);
265  GPIO_setOutputLowOnPin(GPIO_PORT_LED2, GPIO_PIN_LED2);
266  delayMs(LED_ON_TIME);
267  break;
268  }
269  }
270 }

284 void delayUs(uint16_t us)
285 {
286  uint16_t intState;
287  uint16_t count = us/MAX_DELAY_US;
288  if(us)
289  {
290  TB1CTL &= ~(MC_1 | MC_2); //Останавливаем таймер на случай, если он был включен.
291  TB1CTL |= TBCLR; //Сбрасываем TB0 и делители.
292  TB1CCR0 = (MAX_DELAY_US - 1);
293  TB1CTL |= MC__UP; //Запускаем таймер в UP mode.
294  TB1CCTL0 &= ~CCIFG;
295  do
296  {
297  ENTER_CRITICAL_SECTION(intState);
298  while(!(TB1CCTL0 & CCIFG));
299  TB1CCTL0 &= ~CCIFG;
300  EXIT_CRITICAL_SECTION(intState);
301  }while(count--);
302  TB1CTL &= ~(MC_1 | MC_2);
303  }
304 }
